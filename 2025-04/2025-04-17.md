def search_by_tags
  tag_names = params[:tags].to.s.split(',').map(&:split)
  @post = Post.joins(:tags).where(tags: { name: tag_names }).distinct

def search_by_tags

  tag_names = params[:tags].to_s.split(',').map(&:strip)

  @posts = Post.joins(:tags).where(tags: { name: tag_names }).distinct

  if @posts.empty?

    flash.now[:alert] = '該当する投稿はありません'

  end

  render :index

end
