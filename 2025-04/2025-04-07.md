## JS
### callback関数
関数の引数のなかに、また別の関数をいれること
```
function foo(callback) {
    console.log("hi,ken");
    callback();
}

function bar() {
    console.log("hi,there");
}

foo(bar);
```
"hi,ken"と"hi,there"がどちらも出力される

```
function getSeason() {
    const month = new Date().getMonth() + 1;
    if (month >= 3 && month <= 5 ) {
        return "spring";
    } else if (month >= 6 && month <= 8) {
        return "summer";
    } else if (month >= 9 && month <= 11) {
        return "fall";
    } else {
        return "winter";
    }
}

const season = getSeason();
console.log(season);
```
